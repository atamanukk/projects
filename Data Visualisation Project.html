panel.cor <- function(x, y, digits=2, prefix="", cex.cor) {
  usr <- par("usr"); on.exit(par(usr))
  par(usr=c(0,1,0,1))
  r = cor(x, y)
  txt <- format(c(r, 0.123456789), digits=digits)[1]
  txt <- paste(prefix, txt, sep="")
  if (missing(cex.cor)) cex <- 0.8/strwidth(txt)
  text(0.5, 0.5, txt, cex=cex*abs(r))
}

panel.hist <- function(x, col = "#F08080", ...) {
  usr <- par("usr"); on.exit(par(usr))
  par(usr = c(usr[1:2], 0, 1.5))
  h <- hist(x, plot = FALSE)
  breaks <- h$breaks; nB <- length(breaks)
  y <- h$counts; y <- y / max(y)
  rect(breaks[-nB], 0, breaks[-1], y, col = col, ...)
}

Beeswarm Plot

library(ggplot2)
library(ggbeeswarm)
## Warning: package 'ggbeeswarm' was built under R version 4.4.1
ggplot(sleep, aes(x = Sleep.Disorder, y = Sleep.Duration, color = Sleep.Disorder)) +
  geom_beeswarm(size = 3, alpha = 0.7) +
  stat_summary(fun = median, geom = "point", color = "black", size = 1) +
  stat_summary(fun = median, geom = "line", aes(group = 1), color = "black", size = 0.5) +
  scale_color_manual(values = c("None" = "#00AFBB", "Insomnia" = "palegreen3", "Sleep Apnea" = "orange2")) +
  theme_minimal() +
  labs(title = "Uyku Bozukluguna Gore Uyku Suresi",
       x = "",
       y = "Uyku Suresi")

Bu grafik uyku bozukluklarına göre uyku sürelerinin dağılımını göstermek için yapıldı.

  • Varyansların homojen olmadığını, gruplara göre farklılık gösterdiğini söylemek mümkündür.
  • Herhangi bir uyku rahatsızlığına sahip olmayan bireyleri içeren grubun ortalama uyku süresinin diğer gruplara göre daha fazla olduğu ve gruplar arasında anlamlı bir farklılık olduğu görülmektedir.
  • Bir diğer göze çarpan bilgi ise “Uykusuzluk” rahatsızlığını belirten grupta çok sayıda aykırı değerin olduğu göze çarpmaktadır. Aynı zamanda bu grupta değişimin en az ve “Uyku Apnesi” grubunda ise değişimin en fazla olduğu söylenebilir.
